void connectToWiFi() {
  int status = WL_IDLE_STATUS;
  while (status != WL_CONNECTED) {
    status = WiFi.begin(WIFI_SSID, WIFI_PASSWORD);
    delay(1000);
  }
  Serial.println();
  Serial.print("SSID: ");
  Serial.println(WIFI_SSID);
  Serial.print("IP Address: ");
  Serial.println(WiFi.localIP());
}

void sendPostRequest(String data) {
  Serial.print(data);
  if (client.connectSSL(HOST_NAME, 443)) {
    client.print("POST " + PATH_NAME + " HTTP/1.1\r\n");
    client.print("Host: " + String(HOST_NAME) + "\r\n");
    //client.print("Content-Type: application/x-www-form-urlencoded\r\n");
    client.print("Content-Type: application/json\r\n");
    client.print("Content-Length: " + String(data.length()) + "\r\n");
    client.print("Connection: close\r\n");
    client.print("\r\n");
    client.print(data);
    Serial.println("Request sent");

    while (client.connected()) {
      if (client.available()) {
        char c = client.read();
        Serial.print(c);  // Print the response character by character
      }
    }

    // Close the connection
    client.stop();
  } else {
    Serial.println("Connection failed");
  }
}
